<script>
    let thankYouTimerInterval;
    let thankYouTimeLeft = 30 * 60;

    function openThankYouModal() {
        document.getElementById('thankYouModal').classList.add('active');
        document.body.style.overflow = 'hidden';
        
        thankYouTimeLeft = 30 * 60;
        document.getElementById('thankYouMainMessage').style.display = 'block';
        document.getElementById('thankYouSecondMessage').style.display = 'none';
        
        startThankYouTimer();
    }

    function closeThankYouModal() {
        document.getElementById('thankYouModal').classList.remove('active');
        document.body.style.overflow = '';
        
        if (thankYouTimerInterval) {
            clearInterval(thankYouTimerInterval);
        }
    }

    function startThankYouTimer() {
        if (thankYouTimerInterval) {
            clearInterval(thankYouTimerInterval);
        }

        updateThankYouTimerDisplay();

        thankYouTimerInterval = setInterval(() => {
            thankYouTimeLeft--;
            
            if (thankYouTimeLeft <= 0) {
                clearInterval(thankYouTimerInterval);
                showThankYouSecondMessage();
                return;
            }
            
            updateThankYouTimerDisplay();
        }, 1000);
    }

    function updateThankYouTimerDisplay() {
        const minutes = Math.floor(thankYouTimeLeft / 60);
        const seconds = thankYouTimeLeft % 60;
        const display = String(minutes).padStart(2, '0') + ':' + String(seconds).padStart(2, '0');
        document.getElementById('thankYouTimerDisplay').textContent = display;
    }

    function showThankYouSecondMessage() {
        document.getElementById('thankYouMainMessage').style.display = 'none';
        document.getElementById('thankYouSecondMessage').style.display = 'block';
    }

    document.getElementById('thankYouModal').addEventListener('click', function(e) {
        if (e.target === this) {
            closeThankYouModal();
        }
    });

    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') {
            closeThankYouModal();
        }
    });
</script>
